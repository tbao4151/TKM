\section*{DANH MỤC CÁC CHỮ VIẾT TẮT}
\phantomsection \addcontentsline{toc}{section}{\numberline {} DANH MỤC CÁC CHỮ VIẾT TẮT}

\begin{longtable}{l p{12cm}}
    \textbf{CE} & Customer Edge router (Router biên phía khách hàng) \\
    \textbf{CLI} & Command Line Interface (Giao diện dòng lệnh) \\
    \textbf{ECMP} & Equal-Cost Multi-Path (Định tuyến nhiều đường chi phí bằng nhau) \\
    \textbf{ICMP} & Internet Control Message Protocol \\
    \textbf{IP} & Internet Protocol \\
    \textbf{ISP} & Internet Service Provider (Nhà cung cấp dịch vụ) \\
    \textbf{LAN} & Local Area Network (Mạng cục bộ) \\
    \textbf{LDP} & Label Distribution Protocol (Giao thức phân phối nhãn) \\
    \textbf{LSP} & Label Switched Path (Đường chuyển mạch nhãn) \\
    \textbf{MAN} & Metropolitan Area Network (Mạng đô thị) \\
    \textbf{MPLS} & Multiprotocol Label Switching (Chuyển mạch nhãn đa giao thức) \\
    \textbf{OVS} & Open vSwitch \\
    \textbf{P} & Provider router (Router lõi nhà cung cấp) \\
    \textbf{PE} & Provider Edge router (Router biên nhà cung cấp) \\
    \textbf{TCP} & Transmission Control Protocol \\
    \textbf{UDP} & User Datagram Protocol \\
    \textbf{VPLS} & Virtual Private LAN Service \\
\end{longtable}
