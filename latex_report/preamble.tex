% 1. Định dạng cho các chương CÓ ĐÁNH SỐ (Ví dụ: CHƯƠNG I. GIỚI THIỆU)
\titleformat{\section}[block] % Đổi [display] thành [block] để nằm cùng dòng
  {\fontsize{16pt}{0pt}\selectfont \bfseries} % Bỏ \centering để căn trái
  {CHƯƠNG \Roman{section}.}
  {1em} % Khoảng cách giữa chữ "CHƯƠNG I." và Tên chương
  {}
\titlespacing*{\section}{0pt}{0pt}{15pt}

% 2. Định dạng cho các chương KHÔNG ĐÁNH SỐ (Ví dụ: MỤC LỤC, LỜI CẢM ƠN)
\titleformat{name=\section,numberless}[block] % Đổi [display] thành [block]
  {\fontsize{16pt}{0pt}\selectfont \bfseries} % Bỏ \centering
  {}
  {0em}
  {}

% Định dạng mục con (Subsection) - Giữ nguyên

\titlespacing*{\subsection}{0pt}{6pt}{0pt}

\titleformat*{\subsubsection}{\fontsize{13pt}{0pt}\selectfont \bfseries \itshape}
