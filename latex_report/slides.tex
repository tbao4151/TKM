\section{1. Mở đầu \& Giới thiệu}
\begin{frame}{\tdtulogo 1. Mở đầu \& Giới thiệu}
    \begin{itemize}
        \item \textbf{Bối cảnh:} Nhu cầu kết nối mạng nội bộ giữa các chi nhánh cách xa nhau của doanh nghiệp ngày càng tăng cao, đòi hỏi đường truyền ổn định, bảo mật và hiệu suất cao.
        \item \textbf{Giải pháp:} Sử dụng công nghệ lõi \textbf{MPLS} (Multiprotocol Label Switching) trên nền tảng Metro Ethernet.
        \item \textbf{Mục tiêu:} Xây dựng mô hình kết nối 3 chi nhánh doanh nghiệp qua hạ tầng mạng nhà cung cấp.
        \item \textbf{Môi trường mô phỏng:} Mininet, Open vSwitch, và Linux network namespaces.
    \end{itemize}
\end{frame}

\section{2. Thiết kế Kiến trúc mạng LAN}
\begin{frame}{\tdtulogo 2. Thiết kế Kiến trúc mạng LAN}
    Mô hình bao gồm ba chi nhánh với các kiến trúc thiết kế khác nhau:
    \vspace{0.3cm}
    \begin{columns}
        \column{0.33\textwidth}
        \textbf{Branch 1: Flat Network}
        \begin{itemize}
            \item Mạng phẳng đơn giản.
            \item Dành cho quy mô nhỏ.
            \item Tất cả thiết bị chung 1 broadcast domain.
        \end{itemize}
        
        \column{0.33\textwidth}
        \textbf{Branch 2: Three-tier}
        \begin{itemize}
            \item Access - Distribution - Core.
            \item Dành cho quy mô trung bình đến lớn.
            \item Dễ dàng mở rộng, quản trị.
        \end{itemize}
        
        \column{0.33\textwidth}
        \textbf{Branch 3: Leaf-Spine}
        \begin{itemize}
            \item Kiến trúc trung tâm dữ liệu.
            \item Tối ưu lưu lượng East-West.
            \item Độ trễ thấp và dự phòng cao.
        \end{itemize}
    \end{columns}
\end{frame}

\section{3. Thiết kế Mô hình tổng thể (Topology)}
\begin{frame}{\tdtulogo 3. Thiết kế Mô hình tổng thể (Topology)}
    \begin{columns}
        \column{0.45\textwidth}
        \begin{itemize}
            \item \textbf{Mạng nhà cung cấp (Provider Backbone):} 
            \begin{itemize}
                \item \textbf{CE (Customer Edge):} Bộ định tuyến phía khách hàng.
                \item \textbf{PE (Provider Edge):} Bộ định tuyến biên nhà cung cấp.
                \item \textbf{P (Provider Router):} Bộ định tuyến lõi trung tâm.
            \end{itemize}
            \item Lưu lượng liên chi nhánh bắt buộc đi qua lộ trình: CE $\rightarrow$ PE $\rightarrow$ P $\rightarrow$ PE $\rightarrow$ CE.
        \end{itemize}
        \column{0.55\textwidth}
        \centering
        \includegraphics[width=\textwidth]{image/topology_overview.png}
    \end{columns}
\end{frame}

\section{4. Cơ chế hoạt động của MPLS}
\begin{frame}{\tdtulogo 4. Cơ chế hoạt động của MPLS}
    \begin{block}{Quá trình chuyển tiếp bằng nhãn (MPLS Forwarding)}
        Thay vì định tuyến IP qua từng chặng, MPLS chuyển tiếp gói tin nhanh chóng nhờ nhãn:
        \begin{itemize}
            \item \textbf{Ingress PE:} Gán nhãn (Push label) vào gói tin khi đi vào mạng lõi.
            \item \textbf{P Router:} Hoán đổi nhãn (Swap label) và chuyển tiếp ở Layer 2.5, không cần tra bảng định tuyến IP (tránh overhead).
            \item \textbf{Egress PE:} Gỡ nhãn (Pop label) và đẩy gói tin IP chuẩn về mạng đích CE.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{\tdtulogo 4. Minh chứng hoạt động MPLS thực tế}
    \begin{itemize}
        \item Dự án triển khai \textbf{native Linux kernel MPLS}.
        \item Minh chứng thực tế qua \texttt{tcpdump} bắt được gói tin có chứa Header MPLS và ethertype \texttt{0x8847}.
    \end{itemize}

\end{frame}

\section{5. Control-Plane \& L2VPN}
\begin{frame}{\tdtulogo 5. Giao thức điều khiển (Control-Plane) \& L2VPN}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{FRRouting (OSPF/LDP)}
        \begin{itemize}
            \item Sử dụng OSPF định tuyến Underlay.
            \item Sử dụng LDP (Label Distribution Protocol) để phân bổ nhãn động trên PE và P router.
        \end{itemize}
        
        \column{0.5\textwidth}
        \textbf{VPLS/L2VPN Data-plane}
        \begin{itemize}
            \item Các chi nhánh kết nối ở Layer 2 thông qua dịch vụ VPLS.
            \item Hiện thực bằng \textbf{Linux GRETAP pseudowire} với tính năng split-horizon để ngăn vòng lặp.
        \end{itemize}
    \end{columns}
\end{frame}

\section{6. Đo lường \& Phân tích hiệu năng}
\begin{frame}{\tdtulogo 6. Kịch bản Đo lường \& Phân tích hiệu năng}
    Sử dụng các công cụ \textbf{ping, traceroute, iperf3} để đánh giá 4 chỉ số chính:
    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Thông lượng (Throughput):} Khả năng truyền tải tối đa.
        \item \textbf{Độ trễ (Delay / RTT):} Thời gian gói tin đi và về.
        \item \textbf{Độ biến thiên trễ (Jitter):} Sự dao động của độ trễ trong truyền tải UDP.
        \item \textbf{Suy hao gói tin (Packet Loss):} Tỉ lệ gói tin bị mất, đặc biệt đo trong kịch bản tăng tải liên tục (UDP Load Sweep).
    \end{itemize}
\end{frame}

            \item Khi tải vượt ngưỡng chịu đựng của backbone (60 Mbps), suy hao gói tin (Packet Loss) bắt đầu xảy ra rõ rệt ở mức 80M và 120M.
            \item Phản ánh tính trung thực của môi trường mô phỏng.
