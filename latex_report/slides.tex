\section{2. Nội dung lý thuyết}
\begin{frame}{\tdtulogo Rubric 2 - Cơ sở lý thuyết}
    \small
    \begin{columns}
        \column{0.52\textwidth}
        \begin{block}{Metro Ethernet và mô hình CE/PE/P}
            CE thuộc phía khách hàng; PE là biên nhà cung cấp; P là router lõi. Metro Ethernet đóng vai trò hạ tầng MAN để nối các LAN chi nhánh qua mạng provider.
        \end{block}
        \begin{block}{MPLS Label Switching}
            Ingress PE \textbf{push} label, P router \textbf{swap} label, egress PE \textbf{pop} label rồi đưa gói về CE đích.
        \end{block}
        \column{0.48\textwidth}
        \begin{itemize}
            \item OSPF tạo underlay để PE/P biết đường đi.
            \item LDP phân phối label động.
            \item VPLS/L2VPN nối CE liên chi nhánh qua pseudowire.
            \item 3 LAN được so sánh: Flat, Three-tier, Leaf-Spine.
        \end{itemize}
        \vspace{0.15cm}
        \scriptsize Minh chứng: \texttt{results/frr\_control\_plane.txt}, \texttt{results/mpls\_routes.txt}
    \end{columns}
\end{frame}

\begin{frame}{\tdtulogo Rubric 4.4 - Kịch bản đo hiệu năng}
    \small
    \begin{itemize}
        \item \textbf{Throughput:} iperf3 TCP giữa các cặp host nội bộ và liên chi nhánh.
        \item \textbf{Delay:} ping RTT trung bình, parse từ output ping thật.
        \item \textbf{Jitter:} iperf3 UDP JSON, lấy jitter theo luồng UDP.
        \item \textbf{Packet loss:} UDP load sweep ở 5M, 20M, 50M, 80M, 120M.
        \item \textbf{Pipeline:} \texttt{performance\_test.py} ghi \texttt{results.csv}; \texttt{plot\_results.py} sinh biểu đồ Matplotlib.
    \end{itemize}
    \vspace{0.2cm}
    \scriptsize
    Lệnh demo từng phần: \texttt{sudo python3 code/performance\_test.py --action throughput --source-host host1 --destination-host user1 --mode mpls}
\end{frame}
