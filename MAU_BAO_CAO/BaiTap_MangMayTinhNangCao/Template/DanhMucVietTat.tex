\phantomsection \addcontentsline{toc}{section}{\numberline {} DANH MỤC CÁC CHỮ VIẾT TẮT}

\begin{longtable}{l p{12cm}}
    \textbf{ACE} & Access Control Entry (Mục nhập kiểm soát truy cập) \\
    \textbf{ACL} & Access Control List (Danh sách điều khiển truy cập) \\
    \textbf{CLI} & Command Line Interface (Giao diện dòng lệnh) \\
    \textbf{DMZ} & Demilitarized Zone (Vùng phi quân sự) \\
    \textbf{HTTP} & Hypertext Transfer Protocol \\
    \textbf{HTTPS} & Hypertext Transfer Protocol Secure \\
    \textbf{ICMP} & Internet Control Message Protocol \\
    \textbf{IP} & Internet Protocol \\
    \textbf{OVS} & Open vSwitch \\
    \textbf{SSH} & Secure Shell \\
    \textbf{TCP} & Transmission Control Protocol \\
    \textbf{UDP} & User Datagram Protocol \\
    \textbf{VLAN} & Virtual Local Area Network (Mạng cục bộ ảo) \\
\end{longtable}
